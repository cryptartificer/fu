# Histogram Performance Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Eliminate per-line String allocations from histogram pipeline using byte-level parsing with fast-float, memchr, and mmap.

**Architecture:** New `src/hist.rs` module with a two-pass byte-level parser. File input mmap'd, stdin buffered as raw bytes. Existing code untouched — serves as test oracle.

**Tech Stack:** Rust, fast-float, memchr, memmap2, criterion benchmarks

**Baseline (1M rows):** e2e 64ms, of which 41ms is String allocation overhead.

use memchr::memchr_iter;

use crate::data::{self, BarData};

/// Extract the first field from a line, trimming ASCII whitespace.
fn first_field(line: &[u8], delimiter: u8) -> &[u8] {
    let field = match memchr::memchr(delimiter, line) {
        Some(pos) => &line[..pos],
        None => line,
    };
    // Trim ASCII whitespace
    let start = field.iter().position(|&b| b != b' ' && b != b'\t' && b != b'\r').unwrap_or(field.len());
    let end = field.iter().rposition(|&b| b != b' ' && b != b'\t' && b != b'\r').map_or(start, |p| p + 1);
    &field[start..end]
}

/// Fast histogram pipeline: two-pass byte-level parsing.
/// Produces identical output to the existing read_hist_input → filter → bin pipeline.
pub fn hist_from_bytes(
    bytes: &[u8],
    delimiter: u8,
    has_headers: bool,
    gt: Option<f64>,
    lt: Option<f64>,
    log_scale: bool,
    nbins: usize,
) -> Result<BarData, String> {
    // Build line iterator: byte offsets of each line
    let mut line_starts: Vec<usize> = vec![0];
    for pos in memchr_iter(b'\n', bytes) {
        line_starts.push(pos + 1);
    }

    let mut start_idx = 0;
    if has_headers {
        if line_starts.len() <= 1 && bytes.is_empty() {
            return Err("no data (expected header row)".to_string());
        }
        start_idx = 1;
    }

    // Pass 1: parse values, find min/max (and collect values for binning)
    // For correctness we collect into Vec<f64> — the win is avoiding Vec<String>.
    let mut values = Vec::new();
    for i in start_idx..line_starts.len() {
        let line_start = line_starts[i];
        let line_end = if i + 1 < line_starts.len() {
            line_starts[i + 1] - 1 // exclude \n
        } else {
            bytes.len()
        };
        if line_start >= line_end {
            continue; // empty line
        }
        let line = &bytes[line_start..line_end];
        // Strip trailing \r
        let line = if line.last() == Some(&b'\r') {
            &line[..line.len() - 1]
        } else {
            line
        };
        if line.is_empty() {
            continue;
        }

        let field = first_field(line, delimiter);
        if field.is_empty() {
            continue;
        }

        let v: f64 = fast_float::parse(field).map_err(|_| {
            format!(
                "cannot parse as number: {:?}",
                String::from_utf8_lossy(field)
            )
        })?;

        // Apply filter
        if let Some(lo) = gt {
            if v <= lo {
                continue;
            }
        }
        if let Some(hi) = lt {
            if v >= hi {
                continue;
            }
        }

        values.push(v);
    }

    if values.is_empty() {
        return Err("no data".to_string());
    }

    if log_scale {
        data::bin_values_log(&values, nbins)
    } else {
        Ok(data::bin_values(&values, nbins))
    }
}

/// Read from a file using mmap and run the fast histogram pipeline.
pub fn hist_from_file(
    path: &str,
    delimiter: u8,
    has_headers: bool,
    gt: Option<f64>,
    lt: Option<f64>,
    log_scale: bool,
    nbins: usize,
) -> Result<BarData, String> {
    let file = std::fs::File::open(path).map_err(|e| format!("{path}: {e}"))?;
    let mmap = unsafe { memmap2::Mmap::map(&file) }.map_err(|e| format!("{path}: mmap: {e}"))?;
    hist_from_bytes(&mmap, delimiter, has_headers, gt, lt, log_scale, nbins)
}

/// Read from stdin into a byte buffer and run the fast histogram pipeline.
pub fn hist_from_stdin(
    delimiter: u8,
    has_headers: bool,
    gt: Option<f64>,
    lt: Option<f64>,
    log_scale: bool,
    nbins: usize,
) -> Result<BarData, String> {
    use std::io::Read;
    let mut buf = Vec::new();
    std::io::stdin()
        .lock()
        .read_to_end(&mut buf)
        .map_err(|e| e.to_string())?;
    hist_from_bytes(&buf, delimiter, has_headers, gt, lt, log_scale, nbins)
}
